### Installation

Copy and paste the following code into your project.

```js
// tailwind.config.js
module.exports = {
  theme: {
    extend: {
      animation: {
        "modal-fade-in": "modal-fade-in 500ms ease-out",
        "modal-fade-out": "modal-fade-out 500ms ease-in",
      },
      keyframes: {
        "modal-fade-in": {
          "0%": { opacity: "0" },
          "100%": { opacity: "1" },
        },
        "modal-fade-out": {
          "0%": { opacity: "1" },
          "100%": { opacity: "0" },
        },
      },
    },
  },
};
```

components/magicui/video-modal.tsx
